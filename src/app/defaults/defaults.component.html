Setting App-Wide Defaults

All properties set on these components can be set as a default in your importing module. This is mostly useful for properties that you absolutely want to set app-wide, like textAlignLeft or errorHighlightOnTouch on Form Grids, or even toolTipLocation on Form Inputs.

You can also adjust 2 colors:

  • primaryColor - A color which (currently) only changes the color of the icon on a questionText tooltip. (The color of your label text depends on the Form Grid's parent)
  • errorColor - The color of the text and input highlight when there is an ng-invalid form item and errorHighlight is true

To set these defaults, you must pass in a default options object in your module's providers:

import {{ "{" }}
  McFormGridModule,
  FORM_GRID_OPTIONS,
  COLOR_OPTIONS,
} from 'mc-form-grid';

// .... Module Here
providers: [
    {{ '{' }}
      provide: FORM_GRID_OPTIONS,
      useValue:  {{ '{' }}
        boldLabel: true,
        textAlignLeft: true,
        errorHighlightOnTouched: false,
        appendColon: true,
      {{ '}' }},
    {{ '}' }},
    {{ '{' }}
      provide: COLOR_OPTIONS,
      useValue:  {{ '{' }}
        errorColor: '#ff4010',
        primaryColor: '#212529',
      {{ '}' }},
    {{ '}' }},
 ],

In the above example, we importing FORM_GRID_OPTIONS and COLOR_OPTIONS from 'mc-form-grid', but we can provide other default value objects.

  • FORM_GRID_OPTIONS
  • FORM_INPUT_OPTIONS
  • FORM_CUSTOM_OPTIONS
  • COLOR_OPTIONS

Below are the types and values that can be passed into these options objects to override the defaults.

interface FormGridOptions {{ '{' }}
  // Width values are best in 'px' but can be others 'em', 'rem', 'vw', '%' etc.
  labelWidth: string;
  labelPadding: string;
  maxLabelWidth: string;
  columns: string;
  boldLabel: boolean;
  appendColon:boolean;
  textAlignLeft: boolean;
  errorHighlightOnTouched: boolean;
  errorHighlight:boolean;
}

interface FormInputOptions {{ '{' }}
  label: string;
  columnSpan: string;
  rowSpan: string;
  labelWidth: string;
  missingInputText: string | string[];
  multiInput: boolean;
  startingCol: string;
  questionText: string;
  questionTitle: string;
  questionPlacement: 'top' | 'left' | 'right' | 'bottom';
  containerBody: boolean;
  errorMsg: string;
  invalidInput: boolean;
  control: FormControl;
  validationMessages:  {{ '{' }} [x: string]: string };
  alignVertical:boolean;
}

interface FormCustomOptions  {{ '{' }}
  label: string;
  noLabel: boolean;
  columnSpan: string;
  labelWidth: string;
  startingCol: string | number;
}

interface ColorOptions {{ '{' }}
  // Hex or rgb() value
  primaryColor:string; 
  errorColor:string;
}
